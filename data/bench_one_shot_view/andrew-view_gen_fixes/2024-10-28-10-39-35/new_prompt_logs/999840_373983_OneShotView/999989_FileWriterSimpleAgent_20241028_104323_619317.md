  __type: "Creature";

  description: string;
  stats: {
    hp: number;

    skills: Skill[];
  };
}

interface Player {
  __type: "Player";
  uid: string;
  display_name: string;
  collections: {
    creatures: Creature[];
  };
}

interface GameUIData {
  entities: {
    player: Player;
    foe: Player;
    player_creature: Creature;
    foe_creature: Creature;
  };
}

      {/* HUD */}

      </nav>

      {/* Battlefield */}

              name={playerCreature.display_name}
              imageUrl="/placeholder-creature.jpg"

            />

    prototype_id: string;

  collections: {
    skills: Skill[];

  display_name: string;
  description: string;

  __type: "Player";
  meta: {
    prototype_id: string;
    category: string;
  };
  collections: {
    creatures: Creature[];
  };
  uid: string;
  display_name: string;
  description: string;
}

  entities: {
    player: Player;
    opponent: Player;
    player_creature: Creature;

            maxHp={opponent_creature.stats.max_hp}

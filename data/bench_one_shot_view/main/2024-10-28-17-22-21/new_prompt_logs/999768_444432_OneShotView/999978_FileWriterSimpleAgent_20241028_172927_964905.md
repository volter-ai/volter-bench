interface Skill {
  uid: string;
  display_name: string;
  description: string;

    base_damage: number;

    is_physical: boolean;
  };
}

interface Player {
  uid: string;
  display_name: string;
}

interface GameUIData {
  entities: {
    player: Player;
    opponent: Player;

    opponent_creature: Creature;
  };
}

export function MainGameSceneView(props: { data: GameUIData }) {
  const {
    availableButtonSlugs,
    emitButtonClick

  const { player_creature, opponent_creature } = props.data.entities;

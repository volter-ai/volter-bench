                    </div>

                    {/* Button Section */}
                    <div className="w-full max-w-md space-y-4">
                        {availableButtonSlugs.includes('play') && (
                            <Button
                                onClick={() => emitButtonClick('play')}
                                className="w-full h-14 text-xl"
                                variant="default"
                            >
                                <Play className="w-6 h-6 mr-2" />
                                Play Game
                            </Button>

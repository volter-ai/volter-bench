          Game Title
        </div>
      </div>

      <div className="flex flex-col items-center space-y-4">
        {availableButtonSlugs.map((slug) => (
          <Button
            key={slug}

            onClick={() => emitButtonClick(slug)}

            disabled={!enabledUIDs.includes(slug)}

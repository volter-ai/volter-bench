  description: string;
  stats: {
    hp: number;
    max_hp: number;
  };

    skills: Skill[];
  };
}

interface Player {
  uid: string;
  display_name: string;
  entities: {
    active_creature: Creature;
  };
}

  entities: {
    player: Player;
    opponent: Player;
  };
  message?: string;
}

    availableButtonSlugs,
    emitButtonClick

  const player = props.data.entities.player;
  const opponent = props.data.entities.opponent;

  return (

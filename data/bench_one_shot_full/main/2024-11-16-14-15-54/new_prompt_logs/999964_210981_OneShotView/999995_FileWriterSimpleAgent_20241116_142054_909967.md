import withClickable from "@/lib/withClickable"

interface BaseProgressProps extends React.HTMLAttributes<HTMLDivElement> {
  uid: string
  value?: number
}

  ({ uid, ...props }, ref) => {
    return <Progress {...props} ref={ref} />
  }

BaseProgress.displayName = "BaseProgress"

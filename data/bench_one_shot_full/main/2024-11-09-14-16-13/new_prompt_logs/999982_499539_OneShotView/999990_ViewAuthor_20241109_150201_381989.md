  description: string
  stats: {

  }
}

interface Creature {
  __type: "Creature"

  stats: {
    hp: number
    max_hp: number
  }
  collections: {
    skills: Skill[]
  }

  __type: "Player"
  uid: string
  display_name: string

    creatures: Creature[]

    bot_creature: Creature
  }
}
